def calculate_ltv(avg_purchase_value, purchase_frequency, 
                  customer_lifespan, retention_rate, discount_rate=0.1):
    """
    –†–∞—Å—á–µ—Ç –¥–∏—Å–∫–æ–Ω—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ LTV —Å —É—á–µ—Ç–æ–º retention rate
    """
    ltv = 0
    for year in range(1, int(customer_lifespan) + 1):
        year_value = (avg_purchase_value * purchase_frequency * 
                     (retention_rate/100) ** (year - 1))
        discounted_value = year_value / ((1 + discount_rate) ** (year - 1))
        ltv += discounted_value
    return ltv
